# пример - на примере класс Building наследуеться School
 


class Building:
    year = None
    city = None
    size = None
    material = None

    def __init__(self, year, city, size, material):
        self.year = year
        self.city = city
        self.size = size
        self.material = material

    def get_info(self):
        print("Year:", self.year)
        print("City:", self.city)
        print("Size:", self.size)
        print("Material:", self.material)

class School(Building):
     def __init__(self, pupils, classes, year, city, size, material):
        super(School, self).__init__(year, city, size, material)
        self.pupils = pupils
        self.classes = classes

    def get_info(self):
        super(School, self).get_info()
        print("Pupils:", self.pupils, "Classes:", self.classes)

# URL с проверкой на точность являеться ли он ссылкой



import webbrowser

def validator(func):
    def wrapper(url):
        if "." in url:
            func(url)
        else:
            print("Invalid URL")
    return wrapper

@validator

def open_url(url):
    webbrowser.open(url)

open_url("https://EmilNRZS.github.io")
